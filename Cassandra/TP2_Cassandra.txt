//Creacion de las tablas
use bdprogramas;
//Tabla de clientes
CREATE TABLE clientes(
    nom         VARCHAR,
    prod        VARCHAR,
    tipo        VARCHAR,
    PRIMARY KEY(nom, prod)
);
//Tabla de banco
CREATE TABLE banco(
    cte         VARCHAR,
    fecha       TIMESTAMP,
    medio       VARCHAR,
    mov         DOUBLE,
    prod        VARCHAR,
    PRIMARY KEY(cte, fecha))
    WITH CLUSTERING ORDER BY(fecha DESC);

//Creacion del keyspace
create keyspace supermercado with replication = { 'class' : 'SimpleStrategy', 'replication_factor' : 1};

//Creacion de tablas
use supermercado;
//Tabla de compras
create table compra(
    clave int,
    sucursal int,
    fecha text,
    cliente text,
    producto text,
    cantidad int,
    tipo text,
    precio_unit double,
    total double,
    primary key(sucursal, fecha, clave))
    with clustering order by (fecha desc);
//Indice sobre cliente de la tabla compra
create index on compra(cliente);
//Tabla de clientes
create table cliente(
    sucursal int,
    fecha text,
    nombre text,
    primary key(sucursal, fecha, nombre));
